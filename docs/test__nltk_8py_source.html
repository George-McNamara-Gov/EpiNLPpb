103 def test_posTag(self):
104 text = 'The-quick brown, fox "jumps" over the lazy dog.'
105 tokens = nv.wordTokeniser(text)
106 tagVector1 = nv.posTag(tokens)
107 text = 'The-quick brown, fox "jumps".'
108 tokens = nv.wordTokeniser(text)
109 tagVector2 = nv.posTag(tokens)
110 self.assertEqual(len(tagVector1), len(tagVector2))
111 self.assertEqual(tagVector1[0],tagVector2[0])
112 self.assertEqual(tagVector1[2],tagVector2[2])
113 self.assertNotEqual(tagVector1[7],tagVector2[7])
114 self.assertEqual(len(tagVector1), c.MAX_TOKENS)
115 self.assertEqual(tagVector1[1],10)
116 self.assertEqual(tagVector1[6],-1)
117 self.assertEqual(tagVector1[9],5)
118 self.assertEqual(tagVector1[20],-2)
