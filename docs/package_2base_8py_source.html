270 def create(self):
271 '''
272 Imports, vectorises and uses data to train ML component. Then makes
273 predictions and constructs attributes.
274
275 Parameters
276 ----------
277 None
278
279 Returns
280 -------
281 None
282 '''
283 self.trainData, testData = self.importer.importData()
284
285 trainFlags = extractFlags(self.trainData)
286 actualFlags = extractFlags(testData)
287
288 print('Vectorising Data...')
289 time0 = startRec()
290 trainVectors, testVectors = self.vectorise.vectorise(self.trainData,
291 testData)
292 self.vectTime, self.vectSpace = stopRec(time0)
293
294 predictedFlags, trainedModel = self.mlearn.trainAndPredict(trainVectors,
295 trainFlags,
296 testVectors)
297
298 times = [self.importer.importTime,
299 self.importer.filterTime,
300 self.importer.trainExtractTime,
301 self.importer.testExtractTime,
302 self.vectTime,
303 self.mlearn.trainingTime,
304 self.mlearn.predictionTime]
305
306 spaces = [self.importer.importSpace,
307 self.importer.filterSpace,
308 self.importer.trainExtractSpace,
309 self.importer.testExtractSpace,
310 self.vectSpace,
311 self.mlearn.trainingSpace,
312 self.mlearn.predictionSpace]
313
314 self.trainedModel = trainedModel
315 self.actualFlags = actualFlags
316 self.predictedFlags = predictedFlags
317 self.times = times
318 self.spaces = spaces
319 print('NLP program Created')

346 def approximateComplexities(self):
347 '''
348 Approximates the time and space complexities of the program.

350 Parameters

352 None
353
354 Returns

356 None
357 '''
358 print('Approximating Complexities...')

424 def annotateDataCSV(self,
425 path : str,
426 textFieldColumnLabels : list,
427 columnLabel : str):
428 '''
429 Write classifications into a CSV file of records.
430
431 Parameters
432 ----------
433 path : str
434 The location of the CSV file to be annotated.
435 textFieldColumnLabels : list
436 The lables of the columns containg the text to be used.
437 columnLabel : str
438 A label for the column of annotations.
439
440 Returns
441 -------
442 None
443 '''
444 currentTextIndices = self.vectorise.textIndices
445 newTextIndices = list(range(0,len(textFieldColumnLabels)))
446 self.vectorise.setTextIndices(newTextIndices)
447 print('Importing text...')
448 outList = pd.read_csv(
449 path,
450 usecols = textFieldColumnLabels,
451 encoding_errors= 'ignore',
452 low_memory= False
453 ).values.tolist()
454 print('Vectorising...')
455 vectors = self.vectorise.vectoriseList(outList)
456 print('Predicting...')
457 flags = self.trainedModel.predict(vectors)
458 print('Annotating file...')
459 csv_file_path = path
460 try:
461 with open(file= csv_file_path, mode= 'r', errors= 'replace') as file:
462 reader = csv.reader(file)
463 data = list(reader)
464 data[0].append(columnLabel)
465 rowIndex = 1
466 for flag in flags:
467 data[rowIndex].append(flag)
468 rowIndex += 1
469 print('Saving file...')
470 with open(csv_file_path, 'w', newline='', errors= 'replace') as file:
471 writer = csv.writer(file)
472 writer.writerows(data)
473 print('Data annotated.')
474
475 self.vectorise.setTextIndices(currentTextIndices)
476 except Exception as ex:
477 print(ex)
478 print(flags.sum())
