331 def demographicCheck(self,
332 record : list) -> bool:
333 '''
334 Checks if a record relates to a patient in the demographic.
335
336 Parameters
337 ----------
338 record : list
339 The record related to the patient to check.
340
341 Returns
342 -------
343 check : bool
344 Whether or not the related patient is in the demographic.
345 '''
346 check = True
347 if self.dateIndex != -1:
348 if isinstance(record[self.dateIndex], pd.Timestamp):
349 check = check and (
350 self.minYear <= getYearExcel(record[self.dateIndex]) <= self.maxYear)
351 else:
352 check = check and (
353 self.minYear <= getYear(record[self.dateIndex]) <= self.maxYear)
354 if self.hospIndex != -1 and self.hospital != 'ALL':
355 if self.hospital == 'CHHS':
356 check = check and record[self.hospIndex] == 'CHHS'
357 if self.hospital == 'CHPB':
358 check = check and record[self.hospIndex] == 'CHPB'
359 check = check and record[self.hospIndex] == self.hospital
360 if self.sexIndex != -1 and self.sex != 'ALL':
361 if self.sex == 'MALE':
362 check = check and int(record[self.sexIndex]) == 1
363 if self.sex == 'FEMALE':
364 check = check and int(record[self.sexIndex]) == 2
365 if self.ageIndex != -1:
366 check = check and (
367 self.minAge <= float(record[self.ageIndex]) <= self.maxAge)
368 return check
