134 def buildVector(self,
135 text : str) -> list:
136 '''
137 Creates a vector, of consistent length, representing some input text.
138
139 Parameters
140 ----------
141 text : str
142 The text to be represented as a vector.
143
144 Returns
145 -------
146 vector : list
147 The vector representing the input text.
148 '''
149 if 'REMOVE_STOPWORDS' in self.preLAChanges:
150 text = n.stopwordRemoval(text)
151 if 'STEMMING' in self.preLAChanges:
152 text = n.stemming(text)
153
154 match self.tokeniser:
155 case 'WORD_TOKENISER':
156 tokens = n.wordTokeniser(text)
157 case 'PUNC_TOKENISER':
158 tokens = n.puncTokeniser(text)
159 case 'TWEET_TOKENISER':
160 tokens = n.tweetTokeniser(text)
161
162 vector = []
163 if 'POS_TAG' in self.tokenLevelLA:
164 vector += n.posTag(tokens)
165
166 if 'KEYWORDS' in self.textLevelLA:
167 vector = vector + b.keyWordCheck(text)
168 if 'ASCII_CONVERSION' in self.textLevelLA:
169 vector += b.asciiConversion(text)
170
171 return vector

196 '''

226 if 'BAG_OF_WORDS_F' == self.corpusLevelLA:

231 )

235 testRecords,
236 self.textIndices
237 )
238 if 'BAG_OF_WORDS_H' == self.corpusLevelLA:
239 trainVecs, testVecs, self.vectoriser = n.bagOfWordsH(

242 self.textIndices

250 outTrainVecs = sparse.csr_matrix(
