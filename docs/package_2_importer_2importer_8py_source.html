254def splitPosAndNegFlags(data : list) -> tuple[list,list]:
255 '''
256 Splits a list of records into a list of positively flagged records and a
257 list of negatively flagged records.
258
259 Parameters
260 ----------
261 data : list
262 The list of records to be split.
263
264 Returns
265 -------
266 posList : list
267 The list of positively flagged records.
268 negList : list
269 The list of negatively flagged records.
270 '''
271 positive = deque([])
272 negative = deque([])
273 index = 0
274 while index < len(data):
275 if data[index][len(data[index]) - 1] == 1:
276 positive.append(data[index])
277 else:
278 negative.append(data[index])
279 index += 1
280 posList = list(positive)
281 negList = list(negative)
282 return (posList, negList)
